#include <stdio.h>

#include <stdlib.h>

#include <string.h>

 

// Define a structure for student records

struct Student {

    char usn[15];

    char name[50];

    int semester;

};

 

// Function to add student records dynamically

 

void addStudentRecords(struct Student *students, int n) {

    for (int i = 0; i < n; i++)

 {

        printf("\nEnter details for student %d\n", i + 1);

        printf("Enter USN: ");

        scanf("%s", students[i].usn);

        printf("Enter Name: ");

        scanf(" %[^\n]s", students[i].name); // Reads full name including spaces

        printf("Enter Semester: ");

        scanf("%d", &students[i].semester);

    }

    printf("\nRecords added successfully!\n");

}

 

// Function to display all student records

 

void displayRecords(struct Student *students, int n)

 {

    if (n == 0) {

        printf("No records available.\n");

        return;

    }

 

    printf("\nStudent Records:\n");

    printf("-----------------------------------------------------\n");

    printf("| %-12s | %-20s | %-5s |\n", "USN", "Name", "Sem");

    printf("-----------------------------------------------------\n");

 

    for (int i = 0; i < n; i++) {

        printf("| %-12s | %-20s | %-5d |\n", students[i].usn, students[i].name, students[i].semester);

    }

 

    printf("-----------------------------------------------------\n");

}

 

// Function to search for a student record by USN

 

void searchByUSN(struct Student *students, int n) {

    if (n == 0) {

        printf("No records available.\n");

        return;

    }

    char searchUSN[15];

    printf("Enter USN to search: ");

    scanf("%s", searchUSN);

    int found = 0;

    for (int i = 0; i < n; i++) {

        if (strcmp(students[i].usn, searchUSN) == 0) {

            printf("\nRecord Found:\n");

            printf("USN: %s\n", students[i].usn);

            printf("Name: %s\n", students[i].name);

            printf("Semester: %d\n", students[i].semester);

            found = 1;

            break;

        }

    }

    if (!found) {

        printf("\nRecord with USN %s not found!\n", searchUSN);

    }

}

// Main function to drive the program

int main() {

    struct Student *students; // Pointer to dynamically allocated student records

    int n, choice;

    printf("Enter the number of students: ");

    scanf("%d", &n);

 

    // Dynamic memory allocation for student records

 

    students = (struct Student *)malloc(n * sizeof(struct Student));

    if (students == NULL) {

        printf("Memory allocation failed!\n");

        return 1;

    }

 

    addStudentRecords(students, n);

    while (1) {

        printf("\nStudent Record Management System\n");

        printf("1. Display All Records\n");

        printf("2. Search Record by USN\n");

        printf("3. Exit\n");

        printf("Enter your choice: ");

        scanf("%d", &choice);

 

        switch (choice) {

            case 1:

                displayRecords(students, n);

                break;

            case 2:

                searchByUSN(students, n);

                break;

            case 3:

                printf("Exiting program...\n");

                free(students); // Free dynamically allocated memory

                return 0;

            default:

                printf("Invalid choice! Please try again.\n");

        }

    }

 

    return 0;

}

 

 

Memory Allocation

 

#include <stdio.h>

#include <stdlib.h>

int main() {

    int *arr, n, i;

    // Ask user for the number of elements

 

    printf("Enter the number of elements: ");

    scanf("%d", &n);

 

    // Dynamically allocate memory for 'n' integers

 

    arr = (int *)malloc(n * sizeof(int));

 

  // Check if memory allocation was successful

 

    if (arr == NULL) {

        printf("Memory allocation failed!\n");

        return 1;

    }

 

  // Input values for the array

 

    printf("Enter %d elements: \n", n);

    for (i = 0; i < n; i++) {

        scanf("%d", &arr[i]);

    }

 

   // Display the elements

 

    printf("You entered: \n");

    for (i = 0; i < n; i++) {

        printf("%d ", arr[i]);

    }

    printf("\n");

 

    // Reallocate memory to hold more elements

 

    printf("Enter new size: ");

    scanf("%d", &n);

    arr = (int *)realloc(arr, n * sizeof(int)); // Reallocating memory

    // Check if reallocation was successful

    if (arr == NULL) {

        printf("Memory reallocation failed!\n");

        return 1;

    }

    // Input more values if the size was increased

    printf("Enter %d more elements: \n", n);

    for (i = 0; i < n; i++) {

        scanf("%d", &arr[i]);

    }

    // Display the updated array

    printf("Updated array: \n");

    for (i = 0; i < n; i++) {

        printf("%d ", arr[i]);

    }

    printf("\n");

 

    // Free the allocated memory

    free(arr);

 

    return 0;

}

 

 

 

 

Write C program to swap 2 numbers using function

#include <stdio.h>

// Function to swap two numbers

void swap(int *x, int *y) {

    int temp = *x;

    *x = *y;

    *y = temp;

}

int main() {

    int a, b;

    // Asking user for input

    printf("Enter two numbers: ");

    scanf("%d %d", &a, &b);

    // Displaying numbers before swap

    printf("Before swap: a = %d, b = %d\n", a, b);

    // Calling the swap function

    swap(&a, &b);

    // Displaying numbers after swap

    printf("After swap: a = %d, b = %d\n", a, b);

    return 0;

}
